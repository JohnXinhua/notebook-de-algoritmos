  \Input{Una secuencia de coeficientes $a_{0},a_{1},\cdots,a_{n-1},a_{n}$ y un valor de $x$.}
  \Output{El valor del polinomio $\sum_{k=0}^n a_{k}x^{k}$ evaluado en $x$}

  \BlankLine
  $i \gets 0$\;
  $y \gets n$\;
  \While{$i \geq 0$}{
    $y \gets a_{i}+x\cdot y$\;
    $i \gets i-1$\;
  }
  \Return $y$\;
  \caption{\textsc{Horner's-Rule}($\langle a_0,a_1,\cdots a_n \rangle ,x$)}
